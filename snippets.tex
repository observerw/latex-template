\subsection{Pseudocode}

\begin{algorithm}
    \caption{Euclid’s algorithm}
    \label{algo:euclid}
    \begin{algorithmic}[1]
        \Procedure{Euclid}{$a,b$}
        \State $r\gets a\bmod b$
        \While{$r\not=0$}
        \State $a\gets b$
        \State $b\gets r$
        \State $r\gets a\bmod b$
        \EndWhile
        \State \textbf{return} $b$
        \EndProcedure
    \end{algorithmic}
\end{algorithm}

Algorithm~\ref{algo:euclid} shows an example.
